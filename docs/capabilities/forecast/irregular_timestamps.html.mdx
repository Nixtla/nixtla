```python
# Read data
# Dates for the weekends are missing
df = pd.read_csv(
    'https://datasets-nixtla.s3.amazonaws.com/pltr.csv',
    usecols=['date', 'Close'],
)

# Forecast
# the frequency is inferred as B, as only business days are represented in the dataset
forecast_df = nixtla_client.forecast(
    df=df,
    h=5,
    time_col='date', 
    target_col='Close',
)
```
