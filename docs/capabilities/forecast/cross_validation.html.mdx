```python
# Read the data
df = pd.read_csv("https://raw.githubusercontent.com/Nixtla/transfer-learning-time-series/main/datasets/air_passengers.csv")

# Cross-validation using two windows
forecast_cv_df = nixtla_client.cross_validation(
    df=df,
    h=12,
    n_windows=2,
    time_col='timestamp',
    target_col="value",
)
```
